```html
<!DOCTYPE html>
<html lang="zh-CN">
<head>
  <meta charset="UTF-8">
  <style>
    .bar-chart {
      display: flex;
      align-items: flex-end;
      justify-content: space-around;
      height: 300px;
      padding: 40px 20px 60px;
      background: linear-gradient(180deg, #f8f9fa 0%, #ffffff 100%);
      border-radius: 12px;
      position: relative;
    }
    
    .bar-container {
      display: flex;
      flex-direction: column;
      align-items: center;
      gap: 10px;
    }
    
    .bar {
      width: 60px;
      background: linear-gradient(180deg, #4F46E5 0%, #3730A3 100%);
      border-radius: 8px 8px 0 0;
      transition: height 0.8s cubic-bezier(0.16, 1, 0.3, 1);
      position: relative;
    }
    
    .bar-value {
      position: absolute;
      top: -25px;
      left: 50%;
      transform: translateX(-50%);
      font-weight: 600;
      color: #1f2937;
      font-size: 14px;
    }
    
    .bar-label {
      font-size: 14px;
      color: #6b7280;
      text-align: center;
    }
    
    /* Y 轴网格线 */
    .grid-lines {
      position: absolute;
      left: 60px;
      right: 60px;
      top: 40px;
      bottom: 60px;
      pointer-events: none;
    }
    
    .grid-line {
      position: absolute;
      left: 0;
      right: 0;
      border-top: 1px dashed #e5e7eb;
    }
    
    .grid-label {
      position: absolute;
      left: -50px;
      transform: translateY(-50%);
      font-size: 12px;
      color: #9ca3af;
    }
  </style>
</head>
<body>
  <div class="bar-chart" id="chart">
    <div class="grid-lines">
      <div class="grid-line" style="bottom: 0%;"><span class="grid-label">0</span></div>
      <div class="grid-line" style="bottom: 25%;"><span class="grid-label">25</span></div>
      <div class="grid-line" style="bottom: 50%;"><span class="grid-label">50</span></div>
      <div class="grid-line" style="bottom: 75%;"><span class="grid-label">75</span></div>
      <div class="grid-line" style="bottom: 100%;"><span class="grid-label">100</span></div>
    </div>
    <!-- Bars will be generated by JS -->
  </div>

  <script>
    const data = {
      labels: ['Q1', 'Q2', 'Q3', 'Q4'],
      values: [85, 72, 90, 65]
    };
    
    const maxValue = Math.max(...data.values);
    const chart = document.getElementById('chart');
    
    data.values.forEach((value, index) => {
      const container = document.createElement('div');
      container.className = 'bar-container';
      
      const heightPercent = (value / maxValue) * 100;
      
      container.innerHTML = `
        <div class="bar" style="height: 0%" data-height="${heightPercent}%">
          <span class="bar-value">${value}</span>
        </div>
        <span class="bar-label">${data.labels[index]}</span>
      `;
      
      chart.appendChild(container);
      
      // 动画
      setTimeout(() => {
        container.querySelector('.bar').style.height = heightPercent + '%';
      }, 100 + index * 100);
    });
  </script>
</body>
</html>
```

## 饼图 (Pie Chart)

### Canvas 饼图

```html
<canvas id="pieChart" width="400" height="400"></canvas>

<script>
function drawPieChart(canvasId, data) {
  const canvas = document.getElementById(canvasId);
  const ctx = canvas.getContext('2d');
  const centerX = canvas.width / 2;
  const centerY = canvas.height / 2;
  const radius = Math.min(centerX, centerY) - 40;
  
  const colors = [
    '#4F46E5', '#10B981', '#F59E0B', '#EF4444', '#8B5CF6'
  ];
  
  const total = data.values.reduce((sum, val) => sum + val, 0);
  let startAngle = -Math.PI / 2;
  
  data.values.forEach((value, index) => {
    const sliceAngle = (value / total) * 2 * Math.PI;
    
    ctx.beginPath();
    ctx.moveTo(centerX, centerY);
    ctx.arc(centerX, centerY, radius, startAngle, startAngle + sliceAngle);
    ctx.closePath();
    ctx.fillStyle = colors[index % colors.length];
    ctx.fill();
    
    // 绘制百分比标签
    const midAngle = startAngle + sliceAngle / 2;
    const labelX = centerX + Math.cos(midAngle) * (radius * 0.7);
    const labelY = centerY + Math.sin(midAngle) * (radius * 0.7);
    
    ctx.fillStyle = '#ffffff';
    ctx.font = 'bold 14px Arial';
    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';
    ctx.fillText(Math.round(value / total * 100) + '%', labelX, labelY);
    
    startAngle += sliceAngle;
  });
  
  // 绘制图例
  const legendY = canvas.height - 30;
  data.labels.forEach((label, index) => {
    const x = 20 + index * 100;
    
    ctx.fillStyle = colors[index % colors.length];
    ctx.fillRect(x, legendY, 16, 16);
    
    ctx.fillStyle = '#374151';
    ctx.font = '12px Arial';
    ctx.textAlign = 'left';
    ctx.fillText(label, x + 24, legendY + 13);
  });
}

// 使用示例
drawPieChart('pieChart', {
  labels: ['产品 A', '产品 B', '产品 C', '产品 D', '其他'],
  values: [35, 25, 20, 15, 5]
});
</script>
```

## 折线图 (Line Chart)

### Canvas 折线图

```html
<canvas id="lineChart" width="600" height="300"></canvas>

<script>
function drawLineChart(canvasId, data) {
  const canvas = document.getElementById(canvasId);
  const ctx = canvas.getContext('2d');
  const padding = 60;
  const width = canvas.width - padding * 2;
  const height = canvas.height - padding * 2;
  
  const maxValue = Math.max(...data.values);
  const minValue = Math.min(...data.values);
  const range = maxValue - minValue;
  
  // 绘制网格线
  ctx.strokeStyle = '#e5e7eb';
  ctx.setLineDash([5, 5]);
  
  for (let i = 0; i <= 4; i++) {
    const y = padding + (height / 4) * i;
    ctx.beginPath();
    ctx.moveTo(padding, y);
    ctx.lineTo(canvas.width - padding, y);
    ctx.stroke();
    
    // Y 轴标签
    const value = Math.round(maxValue - (range / 4) * i);
    ctx.fillStyle = '#9ca3af';
    ctx.font = '12px Arial';
    ctx.textAlign = 'right';
    ctx.fillText(value, padding - 10, y + 4);
  }
  
  // 绘制折线
  ctx.strokeStyle = '#4F46E5';
  ctx.lineWidth = 3;
  ctx.setLineDash([]);
  ctx.beginPath();
  
  data.values.forEach((value, index) => {
    const x = padding + (width / (data.values.length - 1)) * index;
    const y = padding + height - ((value - minValue) / range) * height;
    
    if (index === 0) {
      ctx.moveTo(x, y);
    } else {
      ctx.lineTo(x, y);
    }
  });
  
  ctx.stroke();
  
  // 绘制数据点
  data.values.forEach((value, index) => {
    const x = padding + (width / (data.values.length - 1)) * index;
    const y = padding + height - ((value - minValue) / range) * height;
    
    // 外圈
    ctx.beginPath();
    ctx.arc(x, y, 8, 0, Math.PI * 2);
    ctx.fillStyle = '#ffffff';
    ctx.fill();
    ctx.strokeStyle = '#4F46E5';
    ctx.lineWidth = 3;
    ctx.stroke();
    
    // 内点
    ctx.beginPath();
    ctx.arc(x, y, 4, 0, Math.PI * 2);
    ctx.fillStyle = '#4F46E5';
    ctx.fill();
    
    // X 轴标签
    ctx.fillStyle = '#6b7280';
    ctx.font = '12px Arial';
    ctx.textAlign = 'center';
    ctx.fillText(data.labels[index], x, canvas.height - padding + 25);
  });
  
  // 绘制曲线下面积
  ctx.lineTo(padding + width, padding + height);
  ctx.lineTo(padding, padding + height);
  ctx.closePath();
  ctx.fillStyle = 'rgba(79, 70, 229, 0.1)';
  ctx.fill();
}

drawLineChart('lineChart', {
  labels: ['1 月', '2 月', '3 月', '4 月', '5 月', '6 月'],
  values: [120, 135, 128, 145, 160, 175]
});
</script>
```

## 动画效果

### 柱状图生长动画
```css
@keyframes growUp {
  from { height: 0; }
  to { height: var(--target-height); }
}

.bar {
  animation: growUp 0.8s cubic-bezier(0.16, 1, 0.3, 1) forwards;
}
```

### 折线图绘制动画
```css
@keyframes drawLine {
  from { stroke-dashoffset: 1000; }
  to { stroke-dashoffset: 0; }
}

.line {
  stroke-dasharray: 1000;
  animation: drawLine 1.5s ease-out forwards;
}
```

### 饼图旋转展开
```css
@keyframes revealPie {
  from { transform: scale(0) rotate(-180deg); }
  to { transform: scale(1) rotate(0); }
}

.pie-chart {
  animation: revealPie 0.8s cubic-bezier(0.16, 1, 0.3, 1);
}
```
